```go
import (
	"fmt"
	"os"

	"gorep/internal/config"
	"gorep/internal/engine"

	"github.com/spf13/cobra"
)

var (
	ignoreCase  bool
	fixedString bool
)

var rootCmd = &cobra.Command{
	Use:   "gorep <pattern> <path>",
	Short: "A fast file search CLI for Go",
	Args:  cobra.ExactArgs(2),
	Run: func(cmd *cobra.Command, args []string) {
		cfg := config.SearchConfig{
			Pattern:     args[0],
			Path:        args[1],
			IgnoreCase:  ignoreCase,
			FixedString: fixedString,
		}

		matcher, err := engine.NewMatcher(cfg.Pattern, cfg.FixedString, cfg.IgnoreCase)
		if err != nil {
			fmt.Println("error:", err)
			os.Exit(1)
		}

		match, err := engine.SearchFile(cfg.Path, matcher)
		if err != nil {
			fmt.Println("error:", err)
			os.Exit(1)
		}

		if match == nil {
			fmt.Println("no matches found")
			return
		}

		fmt.Println(match.FilePath)
		for _, line := range match.Lines {
			fmt.Printf("  %d: %s\n", line.LineNum, line.Content)
		}
	},
}

func init() {
	rootCmd.Flags().BoolVarP(&ignoreCase, "ignore-case", "i", false, "ignore case distinctions")
	rootCmd.Flags().BoolVarP(&fixedString, "fixed-string", "F", false, "treat pattern as a literal string")
}

func Execute() {
	if err := rootCmd.Execute(); err != nil {
		fmt.Println(err)
		os.Exit(1)
	}
}
```

# engine.go

```go
package engine

import (
	"io/fs"
	"os"
	"path/filepath"

	"gorep/internal/model"
)

func SearchPath(path string, matcher *Matcher) ([]model.FileMatch, error) {
	info, err := os.Stat(path)
	if err != nil {
		return nil, err
	}

	if !info.IsDir() {
		match, err := SearchFile(path, matcher)
		if err != nil {
			return nil, err
		}
		if match == nil {
			return nil, nil
		}
		return []model.FileMatch{*match}, nil
	}

	var results []model.FileMatch

	err = filepath.WalkDir(path, func(currentPath string, d fs.DirEntry, err error) error {
		if err != nil {
			return err
		}

		if d.IsDir() {
			return nil
		}

		match, err := SearchFile(currentPath, matcher)
		if err != nil {
			return nil
		}
		if match != nil {
			results = append(results, *match)
		}
		return nil
	})
	if err != nil {
		return nil, err
	}

	return results, nil
}
```

# 4.好，继续：把目录搜索升级成并发搜索

你现在已经有了：

- Matcher

  ：决定一行是否匹配

- SearchFile

  ：搜索单个文件

- SearchPath 单线程版

现在我们做这个项目最核心的一步：

- **Walker**：遍历目录，把文件路径送进 `fileCh`

- **Workers**：多个 goroutine 并发读取 `fileCh`，调用

- **Collector**：主流程从 `resultCh` 收集结果

这一步做完后，你就真正进入了你计划里的 **worker pool + channel pipeline**。

# 先理解这一步在干什么

你可以把整个流程想成工厂流水线：

- **Walker**：负责“找活”
- **Worker 1..N**：负责“干活”
- **Collector**：负责“收货”

数据流是：

```
path -> walker -> fileCh -> workers -> resultCh -> main loop
```

```go
package engine

import (
	"io/fs"
	"os"
	"path/filepath"
	"sync"

	"gorep/internal/config"
	"gorep/internal/model"
)

func SearchPath(cfg config.SearchConfig, matcher *Matcher) ([]model.FileMatch, error) {
	info, err := os.Stat(cfg.Path)
	if err != nil {
		return nil, err
	}

	if !info.IsDir() {
		match, err := SearchFile(cfg.Path, matcher)
		if err != nil {
			return nil, err
		}
		if match == nil {
			return nil, nil
		}
		return []model.FileMatch{*match}, nil
	}

	if cfg.Workers <= 0 {
		cfg.Workers = 1
	}

	fileCh := make(chan string, cfg.Workers*2)
	resultCh := make(chan *model.FileMatch, cfg.Workers)

	var wg sync.WaitGroup

	go func() {
		defer close(fileCh)
		walkFiles(cfg.Path, fileCh)
	}()

	for i := 0; i < cfg.Workers; i++ {
		wg.Add(1)

		go func() {
			defer wg.Done()

			for filePath := range fileCh {
				match, err := SearchFile(filePath, matcher)
				if err != nil {
					continue
				}
				if match != nil {
					resultCh <- match
				}
			}
		}()
	}

	go func() {
		wg.Wait()
		close(resultCh)
	}()

	var results []model.FileMatch
	for match := range resultCh {
		results = append(results, *match)
	}

	return results, nil
}

func walkFiles(root string, fileCh chan<- string) {
	filepath.WalkDir(root, func(path string, d fs.DirEntry, err error) error {
		if err != nil {
			return nil
		}

		if d.IsDir() {
			return nil
		}

		fileCh <- path
		return nil
	})
}
```

# 这段代码你一定要真正看懂

## `fileCh := make(chan string, cfg.Workers*2)`

这是“文件任务队列”。

Walker 会把文件路径塞进去：

```go
fileCh <- path
```

Workers 会从里面拿：

```go
for filePath := range fileCh
```

## `resultCh := make(chan *model.FileMatch, cfg.Workers)`

这是“结果队列”。

每个 worker 找到匹配结果后，把结果发进去：

```go
resultCh <- match
```

主流程统一读取。

------

## `go func() { walkFiles(...) }()`

这里单独开一个 goroutine 做 walker。

它的职责只有一个：

- 遍历目录
- 把每个文件路径送进 `fileCh`

遍历完之后：

```go
defer close(fileCh)
```

告诉 workers：

- 没有新任务了
- 你们读完就可以退出

## `for i := 0; i < cfg.Workers; i++`

这里就是 worker pool。

比如 `workers = 8`，这里就开 8 个 goroutine。

每个 worker 都做同一件事：

- 从 `fileCh` 里拿一个文件路径
- 调 SearchFile
- 有结果就发到 `resultCh`

这就叫 **fan-out**。

------

## `var wg sync.WaitGroup`

`WaitGroup` 用来等待所有 worker 都结束。

流程是：

- 每启动一个 worker，就 `wg.Add(1)`
- worker 退出前 `wg.Done()`
- 另一个 goroutine 里 `wg.Wait()`
- 等所有 worker 都结束后，再 `close(resultCh)`

这一步特别关键。

## 为什么必须关闭 `resultCh`

因为主流程在这里：

```go
for match := range resultCh
```

如果你不关闭 `resultCh`，这个循环就不知道什么时候结束，会一直卡住。

所以一定要等：

- 所有 worker 都退出了
- 确认不会再有新结果了

然后再：

```go
close(resultCh)
```

------

## 为什么这里不用 `Mutex`

因为：

- worker 不直接往同一个 `results` 切片里写
- worker 只负责发消息到 `resultCh`
- 最后只有主 goroutine 在 `append(results, ...)`

所以没有多个 goroutine 同时写同一个切片，不需要加锁。

这就是 channel 的好处。

### 2. 更优雅的替代方案（ripgrep 策略）

真正的搜索工具（如 `grep` 或 `rg`）通常采用更直接的方法：**检查前 N 个字节中是否包含 `NULL` 字节（`0x00`）**。

**推荐的高性能写法（无需引入 `http` 包）：**

```go
func isBinary(data []byte) bool {
    // 只检查前 512 字节
    for _, b := range data {
        if b == 0 {
            return true // 发现 NULL 字节，基本判定为二进制
        }
    }
    return false
}
```
